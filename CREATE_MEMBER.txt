CREATE TABLE MEMBER(
	MEMBER_NUM INTEGER PRIMARY KEY,
	ID VARCHAR(20),
	PW VARCHAR(20),
	EMAIL VARCHAR(50),
	NAME VARCHAR(30),
	ADDR VARCHAR(200),
	PIN INTEGER,
	YEAR INTEGER,
	MONTH INTEGER,
	DAY INTEGER,
	HOBBY VARCHAR(100),
	INTRO VARCHAR(1000)
);
